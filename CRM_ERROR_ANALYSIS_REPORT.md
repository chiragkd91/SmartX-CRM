```python
import time

def create_app(config_class=Config):
    app = Flask(__name__)
    
    # Database connection with retry logic
    max_retries = 3
    retry_delay = 2
    
    for attempt in range(max_retries):
        try:
            db.init_app(app)
            break
        except (OperationalError, ConnectionError) as e:
            if attempt == max_retries - 1:
                raise
            time.sleep(retry_delay)
```
